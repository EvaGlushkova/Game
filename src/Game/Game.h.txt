#ifndef GAME_H
#define GAME_H

#include "GameMode.h"

/**
 * @class Game
 * @brief Main game class
 */
class Game {
private:
    GameMode* mode_;  ///< Game mode
    Deck deck_;       ///< Card deck
    int turn_;        ///< Turn number
    bool gameOver_;   ///< Game ended or not

public:
    /**
     * @brief Constructor
     */
    Game();
    
    /**
     * @brief Destructor
     */
    ~Game();
    
    /**
     * @brief Setup game
     * @param mode Game mode
     * @param difficulty AI difficulty
     */
    void setup(const std::string& mode, const std::string& difficulty = "EASY");
    
    /**
     * @brief Start game
     */
    void start();
    
    /**
     * @brief Check if game ended
     * @return true if game over
     */
    bool isOver() const;
};

#endif
