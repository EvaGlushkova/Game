/**
 * @file Card.h
 * @brief Base abstract class for all cards in the farm game
 */

#ifndef CARD_H
#define CARD_H

#include <string>
#include <iostream>

class Card {
public:
    /**
     * @brief Default constructor
     */
    Card() = default;
    
    /**
     * @brief Virtual destructor
     */
    virtual ~Card() = default;

    /**
     * @brief Get the display name of the card
     * @return Card name as string
     */
    virtual std::string getName() const = 0;
    
    /**
     * @brief Get the card type category
     * @return "Crop" or "Animal"
     */
    virtual std::string getType() const = 0;

    /**
     * @brief Get the cost of this card 
     * @return Cost in coins
     */
    virtual int getCost() const = 0;

    /**
     * @brief Get the income of this card
     * @return The income per activation
     */
    virtual int getIncome() const = 0;
    
    /**
     * @brief Get a description of the card for display
     * @return The string with card information
     */
    std::string getDescription() const;

    /**
     * @brief Activate the card to generate income
     * @return Actual coins generated this activation
     */
    virtual int activateIncome() = 0;
};

#endif
